CREATE TABLE IF NOT EXISTS `user`(
`iduser` INT NOT NULL AUTO_INCREMENT,
`login` VARCHAR(45) NULL,
`password` VARCHAR(45) NULL,
PRIMARY KEY (`iduser`));
 
 CREATE TABLE IF NOT EXISTS `news`(
`idnews` INT primary key auto_increment NOT NULL,
`news` VARCHAR(45) NULL,
`admin_id` INT NULL,
`newscol` VARCHAR(45) NULL);
  
 CREATE TABLE IF NOT EXISTS `profile`(
`idprofile` INT auto_increment primary key not NULL,
`name` VARCHAR(45) NOT NULL,
`level` VARCHAR(45) NOT NULL,
`male/female` VARCHAR(45) NULL,
`email` VARCHAR(45) NULL);
 
 CREATE TABLE IF NOT EXISTS `chat`(
`idchat` INT primary key auto_increment NOT NULL,
`name` VARCHAR(45) NOT NULL,
`participants` VARCHAR(45) NULL);
 
 CREATE TABLE IF NOT EXISTS `message`(
`idmessage` INT primary key auto_increment NOT NULL,
`chat_id` VARCHAR(45) NOT NULL,
`text` VARCHAR(45) NOT NULL,
`chat_idchat` INT NOT NULL);

alter table news
add constraint foreign key (idnews) references user (iduser); 

alter table profile 
add constraint foreign key (idprofile) references user (iduser); 

alter table chat 
add constraint foreign key (idchat) references message (idmessage); 

alter table message
add constraint foreign key (idmessage) references user (iduser);
